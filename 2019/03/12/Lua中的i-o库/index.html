– read从当前输入中读取 all line *number

–快速逐行读（按块读）
local file = assert(io.open(“aa.lua”,’r’))
local bufferSize = 2^13 –每行8kb读（最大8kb缓存值）
file:read(bufferSize,”*line”)
file:close()
